# 获取 Windows 主机 IP,WSK的默认网关ip通常就是主机的ip 
![alt text](asset\image.png)
WINIP=$(ip route | awk '/default/ {print $3; exit}')
echo "Windows 主机 IP: $WINIP"
# 测试 Clash HTTP 代理是否通
curl -I --max-time 5 --proxy http://$WINIP:7897 https://www.google.com
# 设置临时代理（当前终端有效）
export http_proxy="http://$WINIP:7897"
export https_proxy="http://$WINIP:7897"
export HTTP_PROXY="http://$WINIP:7897"
export HTTPS_PROXY="http://$WINIP:7897"
export NO_PROXY="localhost,127.0.0.1,::1,.local"
# 配置 conda 使用代理
conda config --set proxy_servers.http  "http://$WINIP:7897"
conda config --set proxy_servers.https "http://$WINIP:7897"
